// 유클리드 호제법으로 최대공약수 구하기
int GCD(int a, int b) {
  if (a < b) {
    int tm = a;
    a = b;
    b = tm;
  }

  int temp;
  while (a % b != 0) {
    temp = a % b;
    a = b;
    b = temp;
  }
  return b;
}

A B C 최소공배수(LCM)
A와 B의 LCM, C == A,B,C의 LCM

LCM = (a*b)/GCD
GCD = LCM/(a*b)
